		\begin{frame}
			\frametitle{Was ist ein Baum?}
			\vspace{5mm}
			\begin{definition}
				Ein Baum ist ein zusammenhängender kreisfreier ungerichteter Graph.
				
				Sie besteht aus Knoten, die durch Kanten verbunden sind.
				
				Jeder Baum hat einen Wurzelknoten, von dem aus alle anderen Knoten erreichbar sind.
				
				Jeder Knoten kann eine beliebige Anzahl von Kindknoten haben.
			\end{definition}

		\end{frame}

		\begin{frame}
			\frametitle{Was ist ein Heap denn nun konkret?}
			\vspace{5mm}
			\begin{definition}
				In einem Heap (Spezialfall von B\"aumen) sind die Elemente so angeordnet, dass das größte (oder kleinste) Element immer an der Wurzel liegt.
				
				Heaps werden häufig verwendet, um das Maximum (oder Minimum) eines bestimmten Satzes von Elementen effizient abzurufen.
			\end{definition}

		\begin{frame} {Laufzeit von Dijkstra}
			\begin{itemize}
				\item Wir m\"ussen |V| \textbf{extract-min} und |E| \textbf{decrease-key}-Operationen durchf\"uhren.
				\item[] % Leerzeile erzeugen
				\item \underline{\textbf{Extract-Min (f\"ur beide):}} $O(log_2|V|)$
				\item[] % Leerzeile erzeugen				
				\item  \underline{\textbf{Decrease-Key:}}
				\begin{itemize}
					\item  \underline{\textbf{Bin\"are Heaps:}} $O(log_2|V|)$
					\item  \underline{\textbf{Fibonacci-Heaps:}} $O(1)$
				\end{itemize}
				\item[] % Leerzeile erzeugen
				\item  \underline{\textbf{Daraus folgt:}}
				\begin{itemize}
					\item  \underline{\textbf{Bin\"are Heaps:}}  $O((|V| + |E|) log_2|V|) = O((|V| \cdot  log_2|V| + |E| \cdot log_2|V|)$
					\item  \underline{\textbf{Fibonacci-Heaps:}} $O(|E| + |V| log_2|V|)$
				\end{itemize}
			\end{itemize}
			
		\end{frame}
		
		\begin{frame}{Untere Schranke f\"ur Extract-Min}
			\begin{theorem}
				Die theoretische Laufzeit von Extract-Min ist eine untere Schranke f\"ur Heaps im Allgemeinen, und damit optimal.
				($\Omega(\log_2 n)$)
			\end{theorem}
			\ifpauseswitch
				\pause
			\fi
			\begin{proof}
				\begin{itemize}
					\item $\Omega(n \cdot \log_2 n)$ ist eine untere Schranke für vergleichsbasierte Sortieralgorithmen (setzen wir als bekannt voraus; siehe \cite{CORMEN17},\cite{OTTMANN11},\cite{SMI21}).
					\item Ein schnellerer Heap würde Heapsort mit $\text{O}(n)$ ermöglichen, was dem widerspricht.
				\end{itemize}
			\end{proof}
		\end{frame}

	\begin{frame}{Was ist mit Extract-Min?}
		\begin{itemize}
			\item Min-Heaps verwenden einen (fast) vollständigen Binärbaum;
			\begin{itemize}
				\item Jede Ebene außer der untersten ist vollständig gefüllt.
				\item Die unterste Ebene wird von links nach rechts befüllt (\cite{OTTMANN11},\cite{CORMEN17}).
				\item Die Wurzel ist das Minimum, und jeder Elternknoten hat stets einen kleineren Wert als seine Kinder.
			\end{itemize}
			\ifpauseswitch
				\pause
			\fi
			\item \underline{\textbf{Problem:}} Die Laufzeit wegen der Korrektur der Eigenschaften
			\item \underline{\textbf{Ausweg:}} Verzicht auf Teile der Heapeigenschaften
		\end{itemize}

	\begin{frame}{Konkretisierung unserer \"Uberlegungen: Der Fibonacci - Heap}
		\begin{itemize}
			\item Gewurzelte Bäume: Min-Heap geordnet, Elternknoten kleiner/gleich Kindknoten
			\item Verzicht auf binären Baum: Erlaubt mehrere Bäume, nicht vollständig/binär
			\ifpauseswitch
				\pause
			\fi
			\item Wurzelliste: Zyklische doppelt verkettete Liste, Verkettung der Wurzeln
			\item Zeiger auf Minimum: Auf Wurzel mit kleinstem Wert
			\item Kindlisten: Analog zu Wurzelliste, schneller Zugriff auf Kinder jedes Knotens
			\item Zudem verketten wir die Knoten beidseitig -> Zugriff auf alle Knoten in Zeit $O(1)$
			\item Markierungen: Möglichkeit zur Markierung bestimmter Knoten im Heap
		\end{itemize}
	\end{frame}

	\section{Zusammenfassung und Ausblick}
		\begin{frame}
			\frametitle{Zusammenfassung und Ausblick}
			\begin{block}{Zusammenfassung}
				\begin{itemize}
					\item Heaps sind eine Implementierung von Priorit\"atswarteschlangen
					\item  Priorit\"atswarteschlangen sind weit verbreitet
					\item Fibonacci-Heaps gut f\"ur Theorie; u.a. Dijkstra wird amortisiert schneller
					\item Praktisch kaum Anwendung aufgrund der Nachteile au\ss er zu Lehrzwecken kaum Anwendung.
				\end{itemize}
			\end{block}
			\ifpauseswitch
				\pause
			\fi
			\begin{block}{Ausblick}
				\begin{itemize}
					\item Weitere Datenstrukturen (\"uberraschenderweise auch von u.a. Tarjan)
					\item Relaxierte Heaps (sogar auch ohne amortisierte Zeit)
					\item Hollow-Heaps (wesentlich einfachere Struktur mit gleicher Laufzeit; siehe \cite{hansen2015hollow})
				\end{itemize}
				(Sie sind teilweise besser; aber verdr\"angen die \"ublichen Datenstrukturen nicht (siehe \cite{lewis2023comparison},\cite{rhyd_lewis_2023_7759164} sowie \cite{MICHAIL2021100869}))
			\end{block}
		\end{frame}
