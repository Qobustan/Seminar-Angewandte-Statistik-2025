\section{Einleitung \& Motivation}

\begin{frame}{Parametrische vs. Nichtparametrische Dichteschätzung (\"Ubersicht)}
	\begin{columns}[T]
		% Parametrisch
		\begin{column}{0.45\textwidth}
			\textbf{Parametrisch (kennen wir schon $\rightarrow$ Stochastik)}
			\begin{itemize}
				\item Annahme: Verteilung ist bekannt!
				\item Wenige Parameter ($\mu, \sigma^2$)
				\item Schätzung durch Parametersch\"atzer wie bspw. Maximum-Likelihood
			\end{itemize}
		\end{column}
		
		\ifpauseswitch
			\pause
		\fi
		
		% Nichtparametrisch
		\begin{column}{0.45\textwidth}
			\textbf{Nichtparametrisch (KDE)}
			\begin{itemize}
				\item Keine feste Form; die Zufallsvariable $X$ muss aber stetig sein, die Dichte $f$ eine ''glatte'' Funktion
				\item Schätzung: Kern (wie bspw. Epanechnikov) sowie Bandbreite $h$
				\item Erfolgt mittels des Kerndichtesch\"atzers
			\end{itemize}
		\end{column}
	\end{columns}
	
	\vspace{1em}

\end{frame}

\begin{frame}
	\frametitle{Einleitung \& Motivation}
	
	\textbf{Warum nichtparametrisch? \cite{BüningTrenkler+1994}}
	\begin{itemize}
		\item Klassische parametrische Verfahren (z.\,B. unter Annahme einer Normalverteilung) sind oft zu starr.
		\begin{itemize}
			\item Starke Kopplung an unsicheren Annahmen!
		\end{itemize}
		\item Bei unbekannter Verteilung bietet die nichtparametrische Schätzung oft bessere Einblicke in die Datenstruktur.
	\end{itemize}
	
	\ifpauseswitch
		\pause
	\fi
	
	\textbf{Zielsetzung:}
	\begin{itemize}
		\item Annäherung an den Wert $f(x)$ an einer Stelle $x$, ohne eine spezifische parametrische Familie vorauszusetzen.
	\end{itemize}
\end{frame}

\begin{frame}{Ausgangssituation}
	Gegeben seien u.u.i.d.\ Zufallsvariablen
	\[
	X_1,\dots,X_n \sim F
	\quad\text{mit unbekannter Dichte } f.
	\]
	
	\vspace{0.4cm}
	\textbf{Zielsetzung:}
	\[
	\text{Schätze } f(x) \text{ oder } m(x)=\mathbb E(Y|X=x)
	\]
	
	\alert{ohne Annahme einer parametrischen Familie.}
\end{frame}

\section{Der Rosenblatt-Schätzer}

\begin{frame}{Empirische Verteilungsfunktion}
\[
F_n(x)=\frac1n\sum_{i=1}^n \mathds{1}_{\{X_i\le x\}}
\]

\begin{itemize}
  \item Konsistent für $F$
  \item Aber: Treppenfunktion
\end{itemize}

\alert{Wie erhalten wir daraus eine Dichte?}
\end{frame}

\begin{frame}
	\frametitle{Der Rosenblatt-Schätzer (Die theoretische Basis)}
	
	\begin{itemize}
		\item \textbf{Idee:} Da $f(x) = F'(x)$, nutzt man den Differenzenquotienten der empirischen Verteilungsfunktion $F_n$.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Der Rosenblatt-Schätzer (Formel)}
	
	\begin{itemize}
		\item \textbf{Der Schätzer:}
		\begin{equation}
			f_n(x) = \frac{F_n(x+h) - F_n(x-h)}{2h}
		\end{equation}
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Der Rosenblatt-Schätzer (Konsistenz)}
	
	\begin{itemize}
		\item \textbf{Konsistenz:} Damit der Schätzer gegen die wahre Dichte konvergiert, muss für $n \to \infty$ gelten:
		\begin{itemize}
			\item Die Bandbreite $h \to 0$ (Bias reduzieren).
			\item $nh \to \infty$ (Varianz reduzieren).
		\end{itemize}
	\end{itemize}
\end{frame}

\section{Das Histogramm}

\begin{frame}
	\frametitle{Das Histogramm (Der Klassiker)}
	
	\begin{itemize}
		\item \textbf{Funktionsweise:} Zerlegung des Datenraums in Boxen (Bins) der Breite $h$. Die Höhe der Box entspricht der relativen Häufigkeit $\frac{n_k}{n}$ normiert durch $h$.
		\item \textbf{Formal:}
		\[
		\hat f_H(x) = \sum_k \frac{n_k}{nh}\mathds{1}_{B_k}(x)
		\]
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Das Histogramm (Probleme)}
	\begin{itemize}
		\item \textbf{Probleme:}
		\begin{itemize} % statt enumerate
			\item Unstetigkeit: Die geschätzte Dichte ist eine Treppenfunktion.
			\item Subjektivität: Abhängig von $x_0$ und $h$.
		\end{itemize}
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Das Histogramm (Optimale Bandbreite)}
	
	\begin{itemize}
		\item \textbf{Optimale Bandbreite ($h$):}
		\begin{itemize}
			\item Scott: $h \approx 3.49 s n^{-1/3}$
			\item Freedman-Diaconis: $h^* = 2(x_{0.75} - x_{0.25})n^{-1/3}$
		\end{itemize}
	\end{itemize}
\end{frame}

\section{Kernschätzer (KDE)}

\begin{frame}
	\frametitle{Grundidee der Kerndichteschätzung}
	
	\begin{itemize}
		\item Jede Beobachtung trägt zur Schätzung an der Stelle $x$ bei
		\item Der Beitrag hängt vom Abstand zu $x$ ab
	\end{itemize}
	
	\vspace{0.5em}
	
	\begin{itemize}
		\item Einführung einer Gewichtungsfunktion (Kernfunktion)
		\item Glättung statt harter Intervallabgrenzung
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Kernschätzer (KDE) - Einführung}
	
	\begin{itemize}
		\item \textbf{Der Schätzer:}
		\begin{equation}
			\hat{f}_n(x) = \frac{1}{nh} \sum_{i=1}^{n} K\left(\frac{x - X_i}{h}\right)
		\end{equation}
	\end{itemize}
		
	\begin{itemize}
		\item $K$: Kernfunktion
		\item $h > 0$: Bandbreite
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Kernschätzer (KDE) - Wahl des Kerns}
	
	\begin{itemize}
		\item \textbf{Kernwahl:} Der Kern muss zu 1 integrieren ($\int K(x) dx = 1$).
		\item Gängige Kerne:
		\begin{itemize}
			\item Rechteck-Kern
			\item Gauß-Kern
			\item Epanechnikov-Kern
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Kernfunktion – Eigenschaften}
	
	\begin{itemize}
		\item $K(x) \ge 0$ (Nicht-Negativität)
		\item $\int K(x)\,dx = 1$ (Normierung)
		\item Symmetrisch um $0$: $K(x) = K(-x)$
		\item $\sigma_K^2 = \int x^2 K(x)\,dx < \infty$ (Endliche Varianz)
	\end{itemize}
	
	\vspace{0.5em}
	
	\begin{itemize}
		\item Bestimmt die Form der Glättung
		\item Einfluss meist geringer als der der Bandbreite
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Kernschätzer (KDE) - Bandbreitenwahl}
	
	\begin{itemize}
		\item \textbf{Bandbreitenwahl (Silverman's Rule):}
		\begin{equation}
			h_{opt} \approx 1.06 \sigma n^{-1/5}
		\end{equation}
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Kerndichteschätzung – Rolle der Bandbreite}
	
	\begin{itemize}
		\item Kleine Bandbreite: geringe Glättung, hohe Varianz
		\item Große Bandbreite: starke Glättung, hoher Bias
	\end{itemize}
	
	\vspace{0.5em}
	
	\begin{itemize}
		\item Zentrale Stellgröße der Kerndichteschätzung
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{KDE als Faltung}
	
	\[
	\hat f_n(x) = \int K_h(x-y)\, dF_n(y),
	\quad
	K_h(t)=\frac{1}{h} K\left(\frac{t}{h}\right)
	\]
	
	\ifpauseswitch
	\pause
	\fi
	
	\[
	\mathbb{E}[\hat f_n(x)] = (K_h * f)(x)
	\]
	
	\alert{Bias = Glättungsfehler der Faltung}
\end{frame}


\begin{frame}
	\frametitle{Bias--Varianz-Tradeoff}
	
	Unter Regularitätsannahmen:
	\[
	\operatorname{Bias}(\hat f_n(x)) = O(h^2),
	\quad
	\operatorname{Var}(\hat f_n(x)) = O\left(\frac{1}{nh}\right)
	\]
	
	\ifpauseswitch
	\pause
	\fi
	
	\[
	\operatorname{MSE}(x) = O(h^4) + O\left(\frac{1}{nh}\right)
	\]
	
	\alert{$h_{\text{opt}} \sim n^{-1/5}$}
\end{frame}


\begin{frame}
	\frametitle{Bezug zum Rosenblatt-Schätzer}
	
	\begin{itemize}
		\item Rosenblatt-Schätzer als Spezialfall
		\item Rechteckkern mit kompakter Unterstützung
	\end{itemize}
	
	\vspace{0.5em}
	
	\begin{itemize}
		\item Kerndichteschätzung verallgemeinert den Ansatz
		\item Liefert stetige Schätzfunktionen
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Vorteile der Kerndichteschätzung}
	
	\begin{itemize}
		\item Stetige und glatte Schätzfunktion
		\item Flexibel einsetzbar
		\item Gute theoretische Eigenschaften
	\end{itemize}
	
	\vspace{0.5em}
	
	\begin{itemize}
		\item Grundlage für weiterführende nichtparametrische Verfahren
	\end{itemize}
\end{frame}


\section{Nichtparametrische Regression}


\begin{frame}
	\frametitle{Nichtparametrische Regression -- Einführung}
	
	\textbf{Nadaraya--Watson-Schätzer:}
	\begin{equation}
		\hat{m}_{NW}(x) =
		\frac{\sum_{i=1}^n K\left(\frac{x - X_i}{h}\right) Y_i}
		{\sum_{i=1}^n K\left(\frac{x - X_i}{h}\right)}
	\end{equation}
	
	\begin{itemize}
		\item Schätzung von $m(x)$ als gewichteter Mittelwert der umliegenden $Y$-Werte.
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Nichtparametrische Regression -- Zielsetzung}
	
	\textbf{Ziel:}
	\begin{itemize}
		\item Schätzung des funktionalen Zusammenhangs zwischen $X$ und $Y$
		\item Ohne Annahme einer konkreten parametrischen Form für $m(x)$
	\end{itemize}
	
	\vspace{0.5em}
	
	\textbf{Unterschied zur linearen Regression:}
	\begin{itemize}
		\item Kein globales lineares Modell
		\item Lokale Anpassung an die Datenstruktur
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Nichtparametrische Regression -- Lokale Glättung}
	
	\textbf{Grundidee:}
	\begin{itemize}
		\item Punkte mit ähnlichen $X$-Werten liefern ähnliche Informationen über $m(x)$
		\item Beobachtungen in der Nähe von $x$ werden stärker gewichtet
	\end{itemize}
	
	\vspace{0.5em}
	
	\textbf{Umsetzung:}
	\begin{itemize}
		\item Gewichtung erfolgt über eine Kernfunktion $K$
		\item Die Bandbreite $h$ bestimmt die Größe der lokalen Umgebung
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Nadaraya--Watson-Schätzer -- Interpretation}
	
	\textbf{Interpretation des Schätzers:}
	\begin{itemize}
		\item $\hat{m}_{NW}(x)$ ist ein gewichteter Mittelwert der $Y_i$
		\item Gewichte hängen vom Abstand zwischen $X_i$ und $x$ ab
	\end{itemize}
	
	\vspace{0.5em}
	
	\textbf{Eigenschaft:}
	\begin{itemize}
		\item Je näher $X_i$ an $x$ liegt, desto größer ist sein Einfluss
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Nichtparametrische Regression -- Rolle der Bandbreite}
	
	\textbf{Einfluss der Bandbreite $h$:}
	\begin{itemize}
		\item Kleines $h$: starke lokale Anpassung, hohe Varianz
		\item Großes $h$: starke Glättung, höherer Bias
	\end{itemize}
	
	\vspace{0.5em}
	
	\textbf{Analogie zur Dichteschätzung:}
	\begin{itemize}
		\item Gleicher Bias-Varianz-Tradeoff wie bei der KDE
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Nichtparametrische Regression -- Vorteile und Grenzen}
	
	\textbf{Vorteile:}
	\begin{itemize}
		\item Hohe Flexibilität
		\item Keine Modellfehlanpassung durch falsche Annahmen
	\end{itemize}
	
	\vspace{0.5em}
	
	\textbf{Grenzen:}
	\begin{itemize}
		\item Wahl der Bandbreite entscheidend
		\item Randprobleme bei kleinen Stichproben
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Zusammenhang zur Kerndichteschätzung}
	
	\textbf{Gemeinsames Prinzip:}
	\begin{itemize}
		\item Verwendung von Kernfunktionen
		\item Lokale Gewichtung von Beobachtungen
	\end{itemize}
	
	\vspace{0.5em}
	
	\textbf{Unterschied:}
	\begin{itemize}
		\item Dichteschätzung: Schätzung von $f(x)$
		\item Regression: Schätzung von $m(x) = \mathbb{E}(Y|X=x)$
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Strukturelle Sicht}
	
	\[
		m(x) = \frac{\int y\, f_{XY}(x,y)\, dy}{f_X(x)}
	\]
	
	\[
		\hat m(x) = \frac{\hat r(x)}{\hat f_X(x)},
		\quad
		\hat r(x) = \frac{1}{nh} \sum_{i=1}^n
		K\left(\frac{x-X_i}{h}\right) Y_i
	\]
	
	\alert{Quotient von Kerndichtesch\"atzern}
\end{frame}


\section{Robuste Lineare Regression}

\begin{frame}
	\frametitle{Robuste Lineare Regression – Zielsetzung}
	
	\textbf{Ziel:}
	\begin{itemize}
		\item Schätzung eines linearen Zusammenhangs zwischen $X$ und $Y$
		\item Geringe Empfindlichkeit gegenüber Ausreißern
	\end{itemize}
	
	\vspace{0.5em}
	
	\textbf{Motivation:}
	\begin{itemize}
		\item Klassische lineare Regression basiert auf quadratischen Fehlern
		\item Einzelne Ausreißer können die Schätzung stark verzerren
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Robuste Lineare Regression – Grundidee}
	
	\textbf{Zentrale Idee:}
	\begin{itemize}
		\item Betrachtung von Steigungen zwischen Beobachtungspaaren
		\item Verzicht auf Minimierung einer globalen Fehlerfunktion
	\end{itemize}
	
	\vspace{0.5em}
	
	\textbf{Konsequenz:}
	\begin{itemize}
		\item Schätzung basiert auf Lageparametern (Median)
		\item Hohe Robustheit gegenüber extremen Werten
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Robuste Lineare Regression – Rolle des Medians}
	
	\textbf{Warum der Median?}
	\begin{itemize}
		\item Unempfindlich gegenüber einzelnen Ausreißern
		\item Repräsentiert die zentrale Tendenz der Steigungen
	\end{itemize}
	
	\vspace{0.5em}
	
	\textbf{Vergleich:}
	\begin{itemize}
		\item Mittelwert: stark ausreißerempfindlich
		\item Median: robust gegenüber extremen Beobachtungen
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Robuste Regression vs. Kleinste Quadrate}
	
	\textbf{Methode der kleinsten Quadrate:}
	\begin{itemize}
		\item Minimiert die Summe der quadrierten Residuen
		\item Sehr empfindlich gegenüber Ausreißern
	\end{itemize}
	
	\vspace{0.5em}
	
	\textbf{Theil- bzw. Theil–Sen-Schätzer:}
	\begin{itemize}
		\item Basieren auf Medianschätzung
		\item Deutlich robuster gegenüber Ausreißern
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Theil–Sen-Schätzer – Eigenschaften}
	
	\textbf{Wichtige Eigenschaften:}
	\begin{itemize}
		\item Hoher Breakdown-Point
		\item Konsistent unter milden Voraussetzungen
		\item Geeignet bei verrauschten Daten
	\end{itemize}
	
	\vspace{0.5em}
	
	\textbf{Einsatzgebiet:}
	\begin{itemize}
		\item Explorative Datenanalyse
		\item Datensätze mit möglichen Ausreißern
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Robuste Lineare Regression – Grenzen}
	
	\textbf{Einschränkungen:}
	\begin{itemize}
		\item Höherer Rechenaufwand bei großen Stichproben
		\item Beschränkt auf lineare Zusammenhänge
	\end{itemize}
	
	\vspace{0.5em}
	
	\textbf{Abgrenzung:}
	\begin{itemize}
		\item Nichtparametrische Regression bietet höhere Flexibilität
		\item Dafür geringere Interpretierbarkeit
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Robuste Lineare Regression (Theil-Schätzer) - Einführung}
	
	\begin{itemize}
		\item \textbf{Theil-Methode I:} Steigung zwischen Paaren $(i, i+n/2)$, Median als Schätzer.
		\item \textbf{Theil-Sen-Schätzer (Methode II):} Median der Steigungen zwischen allen Paaren $i < j$:
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Robuste Lineare Regression (Theil-Schätzer) - Formel}
	
	\begin{equation}
		H_{ij} = \frac{Y_j - Y_i}{X_j - X_i}
	\end{equation}
	\begin{itemize}
		\item Der Schätzer für den Anstieg $\beta$ ist der Median dieser Steigungen.
	\end{itemize}
\end{frame}

\section{Zusammenfassung}

\begin{frame}
	\frametitle{Take-Home Messages}
	
	\begin{itemize}
		\item Nichtparametrisch = lokal + datengetrieben
		\item KDE = Faltung mit empirischem Maß
		\item Bandbreite bestimmt Konvergenz
		\item Regression folgt demselben Prinzip
		\item Robustheit schützt vor Ausreißern
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Fazit und Ausblick }
	
	\begin{itemize}
		\item \textbf{Flexibilität:} Nichtparametrische Methoden passen sich den Daten an.
		\item \textbf{Parameter $h$:} Kritischer Schritt, Balance zwischen Bias und Varianz.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Fazit und Ausblick (Fortsetzung)}
	
	\begin{itemize}
		\item \textbf{Robustheit:} Verfahren wie der Theil-Sen-Schätzer bieten robuste Alternativen zur klassischen Regression, besonders bei Ausreißern.
	\end{itemize}
\end{frame}
