	\begin{frame}{Ausgangssituation und Zielsetzung}
		Gegeben seien unabhängige, identisch verteilte Zufallsvariablen
		\[
		X_1,\dots,X_n \sim F
		\quad\text{mit unbekannter Dichte } f.
		\]
		
		\vspace{0.5cm}
		\textbf{Zielsetzung:}
		\[
		\text{Schätze } f(x) \quad\text{oder}\quad m(x)=\mathbb E(Y|X=x)
		\]
		
		\alert{ohne Annahme einer parametrischen Familie.}
		
		\vspace{0.5cm}
		\textbf{Grundprinzip:} ,,Lass die Daten sprechen'' – Daten bestimmen die Form der Schätzung.
	\end{frame}
	
	\section{Rosenblatt-Schätzer}
	
	\begin{frame}{Empirische Verteilungsfunktion}
		Die empirische Verteilungsfunktion ist gegeben durch:
		\[
		F_n(x)=\frac1n\sum_{i=1}^n \mathds{1}_{\{X_i\le x\}}
		\]
		
		\begin{itemize}
			\itemsep8pt
			\item Die empirische Verteilungsfunktion ist ein konsistenter Schätzer für die Verteilungsfunktion $F$.
			\item Sie ist jedoch eine Treppenfunktion.
		\end{itemize}
		
		\vspace{0.5em}
		\alert{Wie erhalten wir daraus eine Dichte?}
	\end{frame}
	
	\begin{frame}
		\frametitle{Rosenblatt-Schätzer – Grundidee}
		
		\textbf{Grundprinzip:} Da $f(x) = F'(x)$, approximiere die Ableitung durch den symmetrischen Differenzenquotienten:
		\[
		f(x) \approx \frac{F(x+h) - F(x-h)}{2h}.
		\]
		
		\ifpauseswitch
		\pause
		\fi
		
		\vspace{0.5em}
		
		\textbf{Der Schätzer:} Ersetze $F$ durch die empirische Verteilungsfunktion $F_n$:
		\[
		\hat{f}_n(x) = \frac{F_n(x+h) - F_n(x-h)}{2h}.
		\]
		
		\vspace{0.5em}
		
		\textbf{Interpretation:} Die relative Häufigkeit der Beobachtungen in $(x-h, x+h]$, normiert durch $2h$, ergibt ein ,,gleitendes Histogramm''.
	\end{frame}
	
	\begin{frame}
		\frametitle{Rosenblatt-Schätzer – Konsistenz}
		
		\textbf{Konsistenzbedingungen:} Für $n \to \infty$ muss gelten:
		\begin{enumerate}
			\itemsep8pt
			\item $h \to 0$ \quad (Bias reduzieren: Differenzenquotient approximiert Ableitung besser)
			\item $nh \to \infty$ \quad (Varianz reduzieren: genügend Daten im Intervall)
		\end{enumerate}
		
		\vspace{0.5em}
		
		\textbf{Spannungsverhältnis:} $h$ darf nicht zu schnell gegen Null konvergieren.
		
		\vspace{0.5em}
		
		\textbf{Typische Wahl:} $h \approx n^{-\alpha}$ mit $0 < \alpha < 1$
	\end{frame}
	
	\section{Histogramm}
	
	\begin{frame}
		\frametitle{Funktionsweise eines Histogramms}
		
		\begin{itemize}
			\itemsep8pt
			\item \textbf{Funktionsweise:} Zerlegung des Datenraums in Intervalle (Bins) $B_k$ der Breite $h$. Die Höhe entspricht der relativen Häufigkeit $\frac{n_k}{n}$ normiert durch $h$, wobei $n_k$ die Anzahl der Beobachtungen in $B_k$ bezeichnet.
			\item \textbf{Formal ist ein Histogramm gegeben durch:}
			\[
			\hat f_H(x) = \sum_k \frac{n_k}{nh}\mathds{1}_{B_k}(x)
			\]
			\item Das Histogramm kann damit als Dichteschätzer verwendet werden.
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Histogramm – Probleme}
		\begin{itemize}
			\itemsep8pt
			\item \textbf{Unstetigkeit:} Die geschätzte Dichte ist eine Treppenfunktion.
			\item \textbf{Subjektivität:} Das Ergebnis hängt von der Wahl des Startpunkts $x_0$ und der Bandbreite $h$ ab.
		\end{itemize}
	\end{frame}

	\section{Kerndichteschätzer}
	
	\begin{frame}
		\frametitle{Grundidee der Kerndichteschätzung}
		
		\begin{itemize}
			\itemsep10pt
			\item Bei einer Kerndichteschätzung trägt jede Beobachtung zur Schätzung an der Stelle $x$ bei.
			\item Der Beitrag hängt vom Abstand zur Stelle $x$ ab.
			\item Es wird eine Gewichtungsfunktion (Kernfunktion) eingeführt.
			\item Statt harter Intervallabgrenzung erfolgt eine stetige Glättung.
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Kerndichteschätzer}
		
		Der Kerndichteschätzer ist gegeben durch:
		\[
			\hat{f}_n(x) = \frac{1}{nh} \sum_{i=1}^{n} K\left(\frac{x - X_i}{h}\right),
		\]
		wobei $K$ eine Kernfunktion und $h > 0$ die Bandbreite ist.
	\end{frame}
	
	\begin{frame}
		\frametitle{Kerndichteschätzer – Wahl des Kerns}
		
		\textbf{Gängige Kernfunktionen:}
		\begin{itemize}
			\itemsep8pt
			\item \textbf{Rechteck-Kern} (Gleichverteilung) – einfachste Kernfunktion, unstetig
			\item \textbf{Gauß-Kern} – glatt, unbeschränkter Träger
			\item \textbf{Epanechnikov-Kern} – optimal (minimiert IMSE)
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Praktische Empfehlung:} Gauß-Kern meist ausreichend. Bandbreite $h$ ist entscheidender!
	\end{frame}
	
	\begin{frame}
		\frametitle{Gängige Kernfunktionen – Formeln}
		
		\textbf{Rechteck-Kern (Gleichverteilung auf $[-1,1]$):}
		\[
		K(x) = \begin{cases}
			\frac{1}{2} & \text{falls } |x| \le 1 \\
			0 & \text{sonst}
		\end{cases}
		\]
		
		\vspace{0.5em}
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{Gauß-Kern:}
		\[
		K(x) = \frac{1}{\sqrt{2\pi}} \exp\left(-\frac{x^2}{2}\right)
		\]
	\end{frame}
	
	\begin{frame}
		\frametitle{Epanechnikov-Kern}
		
		\textbf{Epanechnikov-Kern:}
		\[
		K(x) = \begin{cases}
			\frac{3}{4}(1-x^2) & \text{falls } |x| \le 1 \\
			0 & \text{sonst}
		\end{cases}
		\]
		
		\vspace{0.5em}
		
		\textbf{Besonderheit des Epanechnikov-Kerns:}
		\begin{itemize}
			\itemsep8pt
			\item \textbf{Theoretisch optimal:} Minimiert den integrierten mittleren quadratischen Fehler (IMSE $= \int \operatorname{MSE}(x) f(x) \, dx$), der die globale Schätzgüte misst.
			\item Parabolische Form, kompakter Träger (effizienter zu berechnen).
			\item Praktisch kaum Unterschied zu Gauß-Kern (Bandbreite $h$ wichtiger!).
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Kernfunktion – Mathematische Anforderungen}
		
		\textbf{Ein Kern $K$ muss erfüllen:}
		
		\begin{enumerate}
			\itemsep8pt
			\item $K(x) \ge 0$ \\
			verhindert negative Wahrscheinlichkeiten
			
			\item $\int K(x)\, \mathrm{d}x = 1$ \\
			stellt sicher, dass $\hat{f}_n$ eine gültige Dichte ist
			
			\item $K(x) = K(-x)$ \\
			vermeidet systematische Verschiebung
			
			\item $\sigma_K^2 = \int x^2 K(x)\, \mathrm{d}x < \infty$ \\
			notwendig für die Theorie
		\end{enumerate}
		
		\vspace{0.5em}
		
		\textbf{Wichtig:} Die Wahl des Kerns hat meist \textit{geringeren} Einfluss als die Bandbreite $h$.
		
	\end{frame}

		\textbf{Datengetriebene Alternativen:}
		\begin{itemize}
			\itemsep8pt
			\item \textbf{Kreuzvalidierung (Cross-Validation):}
			\begin{itemize}
				\item Keine Normalverteilungsannahme nötig.
				\item Minimiert geschätzten integrierten MSE.
				\item Rechenintensiver.
			\end{itemize}
			
			\item \textbf{Plug-In-Methoden:}
			\begin{itemize}
				\item Schätzen unbekannte Größen aus Daten.
				\item Berechnen optimales $h$ aus asymptotischer MSE-Formel.
			\end{itemize}
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Kerndichteschätzung – Rolle der Bandbreite}
		
		\begin{itemize}
			\itemsep10pt
			\item Eine kleine Bandbreite führt zu geringer Glättung und hoher Varianz.
			\item Eine große Bandbreite führt zu starker Glättung und hohem Bias.
			\item Die Bandbreite ist die zentrale Stellgröße der Kerndichteschätzung.
		\end{itemize}
	\end{frame}
	
	\begin{frame}

		\small \textbf{Bimodale Verteilung:} $h$ zu klein $\Rightarrow$ überangepasst, $h$ optimal $\Rightarrow$ erfasst beide Modes, $h$ zu groß $\Rightarrow$ verschmiert Struktur
	\end{frame}
	
	\begin{frame}
		\frametitle{KDE als Faltung}
		
		Der Kerndichteschätzer lässt sich als Faltung schreiben:
		\[
		\hat{f}_n(x) = \int K_h(x-y)\,dF_n(y),
		\quad
		K_h(t)=\frac{1}{h} K\left(\frac{t}{h}\right).
		\]
		
		\ifpauseswitch
		\pause
		\fi
		
		\vspace{0.5em}
		
		Der Erwartungswert ergibt sich als:
		\[
		\mathbb{E}[\hat{f}_n(x)] = (K_h * f)(x).
		\]
		
		\alert{Bias = Glättungsfehler der Faltung}
	\end{frame}
	
	\begin{frame}
		\frametitle{KDE als Faltung – Interpretation}
		
		\textbf{Was bedeutet die Faltung?}
		\begin{itemize}
			\itemsep8pt
			\item Der Erwartungswert ist eine \textit{geglättete} Version von $f$.
			\item Je breiter der Kern (größeres $h$), desto stärker die Glättung.
			\item Feine Strukturen werden ,,verschmiert''.
			\item Dies erklärt, warum die Wahl von $h$ so kritisch ist.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Bias--Varianz-Tradeoff}
		
		Unter Regularitätsannahmen:
		\[
		\operatorname{Bias}(\hat f_n(x)) = O(h^2),
		\quad
		\operatorname{Var}(\hat f_n(x)) = O\left(\frac{1}{nh}\right)
		\]
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{MSE-Zerlegung:}
		\[
		\operatorname{MSE}(x) = \operatorname{Bias}^2 + \operatorname{Var} = O(h^4) + O\left(\frac{1}{nh}\right)
		\]
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{Optimierung:} Minimiere MSE durch Ableiten nach $h$:
		\[
		\frac{d}{dh}\left[C_1 h^4 + \frac{C_2}{nh}\right] = 4C_1 h^3 - \frac{C_2}{nh^2} \stackrel{!}{=} 0
		\]
		
		\alert{$\Rightarrow h_{\text{opt}} \sim n^{-1/5}$ und $\operatorname{MSE}_{\text{opt}} \sim n^{-4/5}$}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Bezug zum Rosenblatt-Schätzer}
		
		\begin{itemize}
			\itemsep8pt
			\item Der Rosenblatt-Schätzer ist ein Spezialfall des Kerndichteschätzers mit Rechteck-Kern und kompaktem Träger.
			\item Die Kerndichteschätzung verallgemeinert diesen Ansatz.
			\item Sie liefert stetige Schätzfunktionen.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Vorteile der Kerndichteschätzung}
		
		\begin{itemize}
			\itemsep8pt
			\item Stetige und glatte Schätzfunktion.
			\item Flexibel einsetzbar.
			\item Gute theoretische Eigenschaften.
			\item Grundlage für weiterführende nichtparametrische Verfahren.
		\end{itemize}
	\end{frame}
	
	
	\section{Nichtparametrische Regression}
	
	
	\begin{frame}
		\frametitle{Nichtparametrische Regression -- Einführung}
		
		\textbf{Nadaraya--Watson-Schätzer:}
		\[
			\hat{m}_{NW}(x) =
			\frac{\sum_{i=1}^n K\left(\frac{x - X_i}{h}\right) Y_i}
			{\sum_{i=1}^n K\left(\frac{x - X_i}{h}\right)}
		\]
		
		Der Nadaraya-Watson-Schätzer schätzt $m(x)$ als gewichteten Mittelwert der umliegenden $Y$-Werte.
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Nichtparametrische Regression -- Zielsetzung}
		
		\textbf{Ziel:}
		\begin{itemize}
			\itemsep8pt
			\item Schätzung des funktionalen Zusammenhangs zwischen $X$ und $Y$.
			\item Ohne Annahme einer konkreten parametrischen Form für $m(x)$.
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Unterschied zur linearen Regression:}
		\begin{itemize}
			\itemsep8pt
			\item Kein globales lineares Modell.
			\item Lokale Anpassung an die Datenstruktur.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Nichtparametrische Regression -- Lokale Glättung}
		
		\textbf{Grundidee:}
		\begin{itemize}
			\itemsep8pt
			\item Punkte mit ähnlichen $X$-Werten liefern ähnliche Informationen über $m(x)$.
			\item Beobachtungen in der Nähe von $x$ werden stärker gewichtet.
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Umsetzung:}
		\begin{itemize}
			\itemsep8pt
			\item Die Gewichtung erfolgt über eine Kernfunktion $K$.
			\item Die Bandbreite $h$ bestimmt die Größe der lokalen Umgebung.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Nadaraya--Watson-Schätzer -- Interpretation}
		
		\textbf{Interpretation des Schätzers:}
		\begin{itemize}
			\itemsep8pt
			\item $\hat{m}_{NW}(x)$ ist ein gewichteter Mittelwert der $Y_i$.
			\item Die Gewichte hängen vom Abstand zwischen $X_i$ und $x$ ab.
			\item Je näher $X_i$ an $x$ liegt, desto größer ist sein Einfluss.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Nichtparametrische Regression -- Rolle der Bandbreite}
		
		\textbf{Einfluss der Bandbreite $h$:}
		\begin{itemize}
			\itemsep8pt
			\item Ein kleines $h$ führt zu starker lokaler Anpassung und hoher Varianz.
			\item Ein großes $h$ führt zu starker Glättung und höherem Bias.
			\item Es gilt derselbe Bias-Varianz-Tradeoff wie bei der KDE.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Nichtparametrische Regression -- Vorteile und Grenzen}
		
		\textbf{Vorteile:}
		\begin{itemize}
			\itemsep8pt
			\item Hohe Flexibilität.
			\item Keine Modellfehlanpassung durch falsche Annahmen.
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Grenzen:}
		\begin{itemize}
			\itemsep8pt
			\item Die Wahl der Bandbreite ist entscheidend.
			\item Randprobleme bei kleinen Stichproben.
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Boundary Bias Problem}
		
		\textbf{Problem an den Rändern:}
		\begin{itemize}
			\itemsep8pt
			\item Der Nadaraya-Watson-Schätzer hat starken Bias an Datengrenzen.
			\item Weniger Datenpunkte in der Umgebung verfügbar.
			\item Asymmetrische Gewichtung führt zu Verzerrung.
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Lösungsansätze:}
		\begin{itemize}
			\item \textbf{Lokale Polynomschätzer:} Anpassung lokaler Polynome (linear, quadratisch)
			\item Reduzieren Randbias erheblich
			\item Verallgemeinerung des Nadaraya-Watson-Ansatzes
		\end{itemize}
	\end{frame}

	\begin{frame}
		\frametitle{Zusammenhang zur Kerndichteschätzung}
		
		\textbf{Gemeinsames Prinzip:}
		\begin{itemize}
			\itemsep8pt
			\item Verwendung von Kernfunktionen.
			\item Lokale Gewichtung von Beobachtungen.
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Unterschied:}
		\begin{itemize}
			\itemsep8pt
			\item Dichteschätzung: Schätzung von $f(x)$.
			\item Regression: Schätzung von $m(x) = \mathbb{E}(Y|X=x)$.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Strukturelle Sicht}
		
		Der Regressionsfunktionswert lässt sich schreiben als:
		\[
		m(x) = \frac{\int y\, f_{XY}(x,y)\, dy}{f_X(x)}.
		\]
		
		\vspace{0.5em}
		
		Der Nadaraya-Watson-Schätzer ergibt sich daraus als Quotient von Kerndichteschätzern:
		\[
		\hat m(x) = \frac{\hat r(x)}{\hat f_X(x)},
		\quad
		\hat r(x) = \frac{1}{nh} \sum_{i=1}^n
		K\left(\frac{x-X_i}{h}\right) Y_i.
		\]
	\end{frame}
	
	
	\section{Robuste Lineare Regression}
	
	\begin{frame}
		\frametitle{Robuste Lineare Regression – Motivation}
		
		\textbf{Verbindung zur nichtparametrischen Statistik:}
		\begin{itemize}
			\itemsep8pt
			\item Wie die KDE verzichtet auch robuste Regression auf starke Verteilungsannahmen.
			\item Beide zielen auf Robustheit durch reduzierte Verteilungsannahmen und lokale Gewichtung von Datenpunkten.
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Ziel:}
		\begin{itemize}
			\itemsep8pt
			\item Schätzung eines linearen Zusammenhangs zwischen $X$ und $Y$.
			\item Geringe Empfindlichkeit gegenüber Ausreißern.
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Problem der klassischen Regression:}
		\begin{itemize}
			\itemsep8pt
			\item Basiert auf quadratischen Fehlern (OLS).
			\item Einzelne Ausreißer können die Schätzung stark verzerren.
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Robuste Lineare Regression – Grundidee}
		
		\textbf{Zentrale Idee:}
		\begin{itemize}
			\itemsep8pt
			\item Betrachtung von Steigungen zwischen Beobachtungspaaren.
			\item Verzicht auf Minimierung einer globalen Fehlerfunktion.
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Konsequenz:}
		\begin{itemize}
			\itemsep8pt
			\item Die Schätzung basiert auf Lageparametern (Median).
			\item Hohe Robustheit gegenüber extremen Werten.
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Robuste Lineare Regression – Rolle des Medians}
		
		\textbf{Warum der Median?}
		\begin{itemize}
			\itemsep8pt
			\item Unempfindlich gegenüber einzelnen Ausreißern.
			\item Repräsentiert die zentrale Tendenz der Steigungen.
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Vergleich:}
		\begin{itemize}
			\itemsep8pt
			\item Mittelwert: stark ausreißerempfindlich.
			\item Median: robust gegenüber extremen Beobachtungen.
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Robuste Regression vs. Kleinste Quadrate}
		
		\textbf{Methode der kleinsten Quadrate:}
		\begin{itemize}
			\itemsep8pt
			\item Minimiert die Summe der quadrierten Residuen.
			\item Sehr empfindlich gegenüber Ausreißern.
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Theil- bzw. Theil–Sen-Schätzer:}
		\begin{itemize}
			\itemsep8pt
			\item Basieren auf Medianschätzung.
			\item Deutlich robuster gegenüber Ausreißern.
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Theil–Sen-Schätzer – Eigenschaften}
		
		\textbf{Wichtige Eigenschaften:}
		\begin{itemize}
			\itemsep8pt
			\item Hoher Breakdown-Point.
			\item Konsistent unter milden Voraussetzungen.
			\item Geeignet bei verrauschten Daten.
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Einsatzgebiet:}
		\begin{itemize}
			\itemsep8pt
			\item Explorative Datenanalyse.
			\item Datensätze mit möglichen Ausreißern.
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Robuste Lineare Regression – Grenzen}
		
		\textbf{Einschränkungen:}
		\begin{itemize}
			\itemsep8pt
			\item Höherer Rechenaufwand bei großen Stichproben.
			\item Beschränkt auf lineare Zusammenhänge.
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Abgrenzung:}
		\begin{itemize}
			\itemsep8pt
			\item Nichtparametrische Regression bietet höhere Flexibilität.
			\item Dafür geringere Interpretierbarkeit.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Theil-Schätzer – Einführung und Formel}
		
		\begin{itemize}
			\itemsep10pt
			\item \textbf{Theil-Methode I:} Steigung zwischen Paaren $(i, i+n/2)$, Median als Schätzer.
			\item \textbf{Theil-Sen-Schätzer (Methode II):} Median der Steigungen $H_{ij}$ zwischen allen Paaren $i < j$:
		\end{itemize}
		
		\vspace{0.5em}
		
		\[
		H_{ij} = \frac{Y_j - Y_i}{X_j - X_i}
		\]
		
		Der Schätzer für den Anstieg $\beta$ ist der Median dieser Steigungen.
	\end{frame}
	
	\section{Zusammenfassung}
	
	\begin{frame}
		\frametitle{Take-Home Messages}
		
		\begin{itemize}
			\itemsep10pt
			\item Nichtparametrisch = lokal + datengetrieben.
			\item KDE = Faltung mit empirischem Maß.
			\item Bandbreite bestimmt Konvergenz.
			\item Regression folgt demselben Prinzip.
			\item Robustheit schützt vor Ausreißern.
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Fazit und Ausblick}
		
		\begin{itemize}
			\itemsep10pt
			\item \textbf{Flexibilität:} Nichtparametrische Methoden passen sich den Daten an.
			\item \textbf{Parameter $h$:} Kritischer Schritt, Balance zwischen Bias und Varianz.
			\item \textbf{Robustheit:} Verfahren wie der Theil-Sen-Schätzer bieten robuste Alternativen zur klassischen Regression, besonders bei Ausreißern.
		\end{itemize}
	\end{frame}
