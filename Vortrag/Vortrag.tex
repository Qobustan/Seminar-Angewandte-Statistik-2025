	\section{Rosenblatt-Schätzer}
	
	\begin{frame}{Empirische Verteilungsfunktion}
		Die empirische Verteilungsfunktion ist gegeben durch:
		\[
		F_n(x)=\frac1n\sum_{i=1}^n \mathds{1}_{\{X_i\le x\}}
		\]
		
		\begin{itemize}
			\itemsep8pt
			\item Die empirische Verteilungsfunktion ist ein konsistenter Schätzer für die Verteilungsfunktion $F$.
			\item Sie ist jedoch eine Treppenfunktion.
		\end{itemize}
		
		\vspace{0.5em}
		\alert{Wie erhalten wir daraus eine Dichte?}
	\end{frame}
	
	\begin{frame}
		\frametitle{Rosenblatt-Schätzer – Grundidee}
		
		\textbf{Grundprinzip:} Da $f(x) = F'(x)$, approximiere die Ableitung durch den symmetrischen Differenzenquotienten:
		\[
		f(x) \approx \frac{F(x+h) - F(x-h)}{2h}.
		\]
		
		\ifpauseswitch
		\pause
		\fi
		
		\vspace{0.5em}
		
		\textbf{Der Schätzer:} Ersetze $F$ durch die empirische Verteilungsfunktion $F_n$
		\[
		\hat{f}_n(x) = \frac{F_n(x+h) - F_n(x-h)}{2h}.
		\]
		
		\vspace{0.5em}
		
		\textbf{Interpretation:} Die relative Häufigkeit der Beobachtungen in $(x-h, x+h]$, normiert durch $2h$, ergibt ein ,,gleitendes Histogramm''.
	\end{frame}
	
	\begin{frame}
		\frametitle{Rosenblatt-Schätzer – Konsistenz}
		
		\textbf{Konsistenzbedingungen:} Für $n \to \infty$ muss gelten:
		\begin{enumerate}
			\itemsep8pt
			\item $h \to 0$ \quad (Bias reduzieren: Differenzenquotient approximiert Ableitung besser)
			\item $nh \to \infty$ \quad (Varianz reduzieren: genügend Daten im Intervall)
		\end{enumerate}
		
		\vspace{0.5em}
		
		\textbf{Spannungsverhältnis:} $h$ darf nicht zu schnell gegen Null konvergieren.
		
		\vspace{0.5em}
		
		\textbf{Typische Wahl:} $h \approx n^{-\alpha}$ mit $0 < \alpha < 1$
	\end{frame}
	
	\section{Histogramm}
	
	\begin{frame}
		\frametitle{Funktionsweise eines Histogramms}
		
		\begin{itemize}
			\itemsep8pt
			\item \textbf{Funktionsweise:} Zerlegung des Datenraums in Intervalle (Bins) $B_k$ der Breite $h$. Die Höhe entspricht der relativen Häufigkeit $\frac{n_k}{n}$ normiert durch $h$, wobei $n_k$ die Anzahl der Beobachtungen in $B_k$ bezeichnet.
			\item \textbf{Formal ist ein Histogramm gegeben durch:}
			\[
			\hat f_H(x) = \sum_k \frac{n_k}{nh}\mathds{1}_{B_k}(x)
			\]
			\item Das Histogramm kann damit als Dichteschätzer verwendet werden.
		\end{itemize}
	\end{frame}

	\section{Kerndichteschätzer}
	
	\begin{frame}
		\frametitle{Kerndichteschätzer}
		
		\begin{itemize}
			\itemsep8pt
			\item Bei einer Kerndichteschätzung trägt jede Beobachtung zur Schätzung an der Stelle $x$ bei.
			\item Der Beitrag hängt vom Abstand von $X_i$ zu $x$ ab – nahe Beobachtungen werden stärker gewichtet.
			\item Statt harter Intervallabgrenzung erfolgt eine stetige Glättung über eine Gewichtungsfunktion (Kernfunktion).
		\end{itemize}
		
		\vspace{0.5em}
		
		Der Kerndichteschätzer ist gegeben durch:
		\[
			\hat{f}_n(x) = \frac{1}{nh} \sum_{i=1}^{n} K\!\left(\frac{x - X_i}{h}\right),
		\]
		wobei $K$ eine Kernfunktion und $h > 0$ die Bandbreite ist.
	\end{frame}
	
	\begin{frame}
		\frametitle{Kerndichteschätzer – Wahl des Kerns}
		
		\textbf{Gängige Kernfunktionen:}
		\begin{itemize}
			\itemsep8pt
			\item \textbf{Rechteck-Kern} (Gleichverteilung) – einfachste Kernfunktion, unstetig
			\item \textbf{Gauß-Kern} – glatt, unbeschränkter Träger
			\item \textbf{Epanechnikov-Kern} – optimal (minimiert IMSE)
		\end{itemize}
		
		\vspace{0.5em}
		
		\textbf{Praktische Empfehlung:} Gauß-Kern meist ausreichend. Bandbreite $h$ ist entscheidender!
	\end{frame}
	
	\begin{frame}
		\frametitle{Gängige Kernfunktionen – Formeln}
		
		\textbf{Rechteck-Kern (Gleichverteilung auf $[-1,1]$):}
		\[
		K(x) = \begin{cases}
			\frac{1}{2} & \text{falls } |x| \le 1 \\
			0 & \text{sonst}
		\end{cases}
		\]
		
		\vspace{0.5em}
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{Gauß-Kern:}
		\[
		K(x) = \frac{1}{\sqrt{2\pi}} \exp\left(-\frac{x^2}{2}\right)
		\]
		
		\vspace{0.5em}
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{Epanechnikov-Kern:}
		\[
		K(x) = \begin{cases}
			\frac{3}{4}(1-x^2) & \text{falls } |x| \le 1 \\
			0 & \text{sonst}
		\end{cases}
		\]
	\end{frame}
	
	\begin{frame}
		\frametitle{Besonderheit des Epanechnikov-Kerns}
		
		\begin{itemize}
			\itemsep10pt
			\item \textbf{Theoretisch optimal:} Der Epanechnikov-Kern minimiert den integrierten mittleren quadratischen Fehler
			\[
			\operatorname{IMSE} = \int \operatorname{MSE}\bigl(\hat{f}_n(x)\bigr)\, f(x) \, dx,
			\]
			der die globale Güte der Schätzung misst.
			\item Parabolische Form mit kompaktem Träger (effizienter zu berechnen).
			\item Praktisch kaum Unterschied zu Gauß-Kern – die Bandbreite $h$ ist entscheidender.
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Kernfunktion – Mathematische Anforderungen}
		
		\textbf{Ein Kern $K$ muss erfüllen:}
		
		\begin{enumerate}
			\itemsep10pt
			\item $K(x) \ge 0$ \qquad (verhindert negative Wahrscheinlichkeiten)
			
			\item $\int K(x)\, \mathrm{d}x = 1$ \qquad (stellt sicher, dass $\hat{f}_n$ eine gültige Dichte ist)
			
			\item $K(x) = K(-x)$ \qquad (vermeidet systematische Verschiebung)
			
			\item $\sigma_K^2 = \int x^2 K(x)\, \mathrm{d}x < \infty$ \qquad (notwendig für die Theorie)
		\end{enumerate}
	\end{frame}
	
	
	
	\begin{frame}
		\frametitle{Bandbreitenwahl}
		
		\textbf{Silvermans Faustformel:}
		\[
		h \approx 1{,}06\,\hat{\sigma}\,n^{-1/5}
		\]
		
		\ifpauseswitch
		\pause
		\fi
		
		\vspace{0.5em}
		
		\textbf{Datengetriebene Alternativen:}
		\begin{itemize}
			\itemsep8pt
			\item \textbf{Plug-In-Methoden:}
			\begin{itemize}
				\normalsize
				\item Schätzen unbekannte Größen aus Daten.
				\item Berechnen optimalen Wert für $h$ aus asymptotischer MSE-Formel.
			\end{itemize}
			
			\item \textbf{Kreuzvalidierung:}
			\begin{itemize}
				\normalsize
				\item Keine Normalverteilungsannahme nötig.
				\item Minimiert geschätzten integrierten MSE.
				\item Rechenintensiver.
			\end{itemize}
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Kerndichteschätzung – Rolle der Bandbreite}
		
		\textbf{Zentrale Stellgröße der Kerndichteschätzung:}
		\begin{itemize}
			\itemsep8pt
			\item Kleine Bandbreite: geringe Glättung, hohe Varianz (überangepasst).
			\item Große Bandbreite: starke Glättung, hoher Bias (überglätten).
		\end{itemize}
		
		\vspace{0.3em}
		
		\centering

	\begin{frame}
		\frametitle{KDE als Faltung}
		
		Der Kerndichteschätzer lässt sich als Faltung schreiben:
		\[
		\hat{f}_n(x) = \int K_h(x-y)\,dF_n(y),
		\qquad
		K_h(t)=\frac{1}{h} K\!\left(\frac{t}{h}\right).
		\]
		
		\ifpauseswitch
		\pause
		\fi
		
		\vspace{0.5em}
		
		Somit ergibt sich der Erwartungswert als:
		\[
		\mathbb{E}\bigl[\hat{f}_n(x)\bigr] = (K_h * f)(x).
		\]
		
		Da der Erwartungswert eine \textit{geglättete} Version von $f$ ist, folgt:
		\begin{itemize}
			\itemsep6pt
			\item Je breiter der Kern (größeres $h$), desto stärker die Glättung.
			\item Feine Strukturen werden verwischt.
			\item \alert{Bias = Glättungsfehler der Faltung}
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{$O$-Notation (angelehnt an \cite{SMI21} sowie \cite{dal2023probtest})}
		
		\begin{columns}[T]
			\begin{column}{0.52\textwidth}
				\textbf{Betrachte Funktionen:}
				\begin{itemize}
					\itemsep6pt
					\item $f : \mathbb{N} \to \mathbb{R}$ \quad (Laufzeit/Fehler)
					\item $g : \mathbb{N} \to \mathbb{R}$ \quad (Wachstumsordnung)
				\end{itemize}
				
				\vspace{0.5em}
				
				$f$ \textbf{wächst höchstens so schnell wie} $g$, falls es $c > 0$ und $n_0 \in \mathbb{N}$ gibt mit
				\[
				0 \leq f(n) \leq c \cdot g(n) \quad \forall\, n \geq n_0.
				\]
				
				\vspace{0.3em}
				
				\textbf{Schreibweise:} $f(n) \in O(g(n))$
			\end{column}
			
			\begin{column}{0.44\textwidth}
				\centering
				\begin{tikzpicture}
					\begin{axis}[
						width=5.5cm, height=4.5cm,
						xlabel={$n$},
						xmin=0, xmax=8,
						ymin=0, ymax=14,
						xtick={4},
						xticklabels={$n_0$},
						ytick=\empty,
						legend pos=north west,
						legend style={font=\tiny},
						domain=0.01:8,
						samples=80,
						axis lines=left,
						]
						\addplot[blue, thick] {0.5*x^2};
						\addlegendentry{$c \cdot g(n)$}
						\addplot[red, thick] {x^1.5};
						\addlegendentry{$f(n)$}
						\draw[dashed, gray, thick] (axis cs:4,0) -- (axis cs:4,8);
						\node[font=\tiny, gray, align=center] at (axis cs:6.2, 3.5)
							{$f(n) \leq c \cdot g(n)$\\[-2pt]$\forall\, n \geq n_0$};
					\end{axis}
				\end{tikzpicture}
			\end{column}
		\end{columns}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Bias--Varianz-Tradeoff}
		
		Unter gewissen Regularitätsannahmen gilt:
		\[
		\operatorname{Bias}\bigl(\hat f_n(x)\bigr) = O(h^2),
		\qquad
		\operatorname{Var}\bigl(\hat f_n(x)\bigr) = O\!\left(\frac{1}{nh}\right)
		\]
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{MSE-Zerlegung:} Mit $c_1, c_2 > 0$ gilt:
		\[
		\operatorname{MSE}\bigl(\hat{f}_n(x)\bigr) = \operatorname{Bias}^2\!\bigl(\hat{f}_n(x)\bigr) + \operatorname{Var}\bigl(\hat{f}_n(x)\bigr) \approx c_1 h^4 + \frac{c_2}{nh}
		\]
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{Optimierung:} Minimiere MSE durch Ableiten nach $h$:
		\[
		\frac{d}{dh}\!\left[c_1 h^4 + \frac{c_2}{nh}\right] = 4c_1 h^3 - \frac{c_2}{nh^2} \stackrel{!}{=} 0
		\]
		
		\alert{$\Rightarrow h_{\text{opt}} \sim n^{-1/5}$ \qquad und \qquad $\operatorname{MSE}_{\text{opt}} \sim n^{-4/5}$}
	\end{frame}
	
	

	
	
	\begin{frame}
		\frametitle{Bezug zum Rosenblatt-Schätzer}
		
		Der Rosenblatt-Schätzer
		\[
		\hat{f}_n(x) = \frac{F_n(x+h) - F_n(x-h)}{2h}
		\]
		kann als Spezialfall des Kerndichteschätzers mit Rechteck-Kern
		\[
		K(u) = \tfrac{1}{2}\,\mathbf{1}_{[-1,1]}(u)
		\]
		aufgefasst werden: Einsetzen in $\frac{1}{nh}\sum_i K\!\bigl(\frac{x-X_i}{h}\bigr)$ liefert genau $\hat{f}_n$.
		
		\vspace{0.5em}
		
		Die Kerndichteschätzung verallgemeinert diesen Ansatz und liefert stetige Schätzfunktionen.
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Vorteile der Kerndichteschätzung}
		
		\textbf{Vorteile der Kerndichteschätzung:}
		\begin{itemize}
			\itemsep10pt
			\item Sie liefert eine stetige und glatte Schätzfunktion.
			\item Sie ist flexibel einsetzbar ohne Annahme einer parametrischen Familie.
			\item Sie hat gute theoretische Eigenschaften (Konsistenz, bekannte Konvergenzraten).
			\item Sie bildet die Grundlage für weiterführende nichtparametrische Verfahren.
		\end{itemize}
	\end{frame}
	
	
	\section{Nichtparametrische Regression}
	
	
	\begin{frame}
		\frametitle{Nichtparametrische Regression – Zielsetzung}
		
		\textbf{Regressionsmodell:} Gegeben seien Beobachtungspaare $(X_1, Y_1), \ldots, (X_n, Y_n)$ mit
		\[
		Y_i = m(X_i) + \varepsilon_i, \qquad \mathbb{E}(\varepsilon_i) = 0.
		\]
		Die \textbf{Regressionsfunktion} $m(x) = \mathbb{E}(Y \mid X = x)$ beschreibt den mittleren Zusammenhang zwischen $X$ und $Y$.
		
		\vspace{0.5em}
		
		\textbf{Ziel:} Schätzung von $m(x)$ ohne Annahme einer konkreten parametrischen Form.
		
		\vspace{0.5em}
		
		\textbf{Unterschied zur linearen Regression:}
		\begin{itemize}
			\itemsep6pt
			\item Kein globales lineares Modell $m(x) = \beta_0 + \beta_1 x$.
			\item Lokale Anpassung an die Datenstruktur.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Nichtparametrische Regression – Grundidee}
		
		\textbf{Grundidee der lokalen Glättung:}
		\begin{itemize}
			\itemsep8pt
			\item Beobachtungen mit ähnlichen Werten für $X$ liefern ähnliche Informationen über $m(x)$.
			\item Beobachtungen in der Nähe von $x$ werden stärker gewichtet.
		\end{itemize}
		
		\vspace{0.5em}
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{Umsetzung:}
		\begin{itemize}
			\itemsep8pt
			\item Die Gewichtung erfolgt über eine Kernfunktion $K$.
			\item Die Bandbreite $h$ bestimmt die Größe der lokalen Umgebung.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Nadaraya--Watson-Schätzer}
		
		Der \textbf{Nadaraya--Watson-Schätzer} schätzt $m(x)$ als gewichteten Mittelwert der $Y_i$:
		\[
			\hat{m}_{NW}(x) =
			\frac{\displaystyle\sum_{i=1}^n K\!\left(\frac{x - X_i}{h}\right) Y_i}
			{\displaystyle\sum_{i=1}^n K\!\left(\frac{x - X_i}{h}\right)}.
		\]
		
		\vspace{0.5em}
		
		\textbf{Interpretation:}
		\begin{itemize}
			\itemsep6pt
			\item $\hat{m}_{NW}(x)$ ist ein gewichteter Mittelwert der $Y_1, \ldots, Y_n$.
			\item Das Gewicht von $Y_i$ hängt vom Abstand von $X_i$ zu $x$ ab.
			\item Je näher $X_i$ an $x$ liegt, desto größer ist sein Einfluss.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Nichtparametrische Regression – Rolle der Bandbreite}
		
		\textbf{Einfluss der Bandbreite $h$:}
		\begin{itemize}
			\itemsep8pt
			\item Ein kleines $h$ führt zu starker lokaler Anpassung und hoher Varianz.
			\item Ein großes $h$ führt zu starker Glättung und höherem Bias.
			\item Es gilt derselbe Bias-Varianz-Tradeoff wie bei der Kerndichteschätzung.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Nichtparametrische Regression – Vorteile und Grenzen}
		
		\textbf{Vorteile der nichtparametrischen Regression:}
		\begin{itemize}
			\itemsep8pt
			\item Hohe Flexibilität – keine Modellfehlanpassung durch falsche Annahmen.
			\item Anwendbar auch wenn die funktionale Form von $m$ unbekannt ist.
		\end{itemize}
		
		\vspace{0.5em}
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{Grenzen:}
		\begin{itemize}
			\itemsep8pt
			\item Die Wahl der Bandbreite ist entscheidend für die Schätzgüte.
			\item Randprobleme: erhöhter Bias an den Grenzen des Datenbereichs.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Bias-Problem an den Datengrenzen}
		
		\textbf{Problem an den Rändern:}
		\begin{itemize}
			\itemsep8pt
			\item Der Nadaraya-Watson-Schätzer weist starken Bias an den Datengrenzen auf.
			\item An den Rändern stehen weniger Datenpunkte in der Umgebung zur Verfügung.
			\item Die asymmetrische Gewichtung führt zur Verzerrung der Schätzung.
		\end{itemize}
		
		\vspace{0.5em}
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{Lösungsansatz – Lokale Polynomschätzer:}
		\begin{itemize}
			\itemsep8pt
			\item Anpassung lokaler Polynome (linear, quadratisch) statt Konstanten.
			\item Reduzieren den Randbias erheblich.
			\item Verallgemeinerung des Nadaraya-Watson-Ansatzes.
		\end{itemize}
	\end{frame}

	\begin{frame}
		\frametitle{
			Gemeinsames Prinzip von Kerndichteschätzung und nichtparametrischer Regression
			(basierend auf \cite{BüningTrenkler+1994})
			}
		
		\textbf{Gemeinsames Prinzip:}
		\begin{itemize}
			\itemsep8pt
			\item Verwendung von Kernfunktionen zur lokalen Gewichtung von Beobachtungen.
			\item Die Bandbreite $h$ steuert den Grad der Glättung in beiden Verfahren.
		\end{itemize}
		
		\vspace{0.5em}
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{Unterschied:}
		\begin{itemize}
			\itemsep8pt
			\item Kerndichteschätzung: Schätzung der Dichte $f(x)$ von $X$.
			\item Nichtparametrische Regression: Schätzung von $m(x) = \mathbb{E}(Y \mid X = x)$.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Strukturelle Sicht – Nadaraya-Watson als Quotient von KDE}
		
		Die Regressionsfunktion lässt sich schreiben als:
		\[
		m(x) = \frac{\int y\, f_{XY}(x,y)\, dy}{f_X(x)}.
		\]
		
		\vspace{0.5em}
		
		Der Nadaraya-Watson-Schätzer ergibt sich daraus als Quotient von Kerndichteschätzern:
		\[
		\hat m(x) = \frac{\hat r(x)}{\hat f_X(x)},
		\qquad
		\hat r(x) = \frac{1}{nh} \sum_{i=1}^n
		K\!\left(\frac{x-X_i}{h}\right) Y_i.
		\]
	\end{frame}
	
	
	\showskipbuttontrue
	\section{Robuste Lineare Regression}
	
	\begin{frame}
		\frametitle{Robuste Lineare Regression – Motivation}
		
		\textbf{Problem der klassischen linearen Regression:}
		\begin{itemize}
			\itemsep8pt
			\item Basiert auf quadratischen Fehlern (Methode der kleinsten Quadrate).
			\item Einzelne Ausreißer können die Schätzung stark verzerren.
		\end{itemize}
		
		\vspace{0.5em}
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{Ziel der robusten linearen Regression:}
		\begin{itemize}
			\itemsep8pt
			\item Schätzung eines linearen Zusammenhangs zwischen $X$ und $Y$.
			\item Geringe Empfindlichkeit gegenüber Ausreißern.
		\end{itemize}
		
		\vspace{0.5em}
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{Verbindung zur nichtparametrischen Statistik:}
		\begin{itemize}
			\itemsep8pt
			\item Wie die Kerndichteschätzung verzichtet auch robuste Regression auf starke Verteilungsannahmen.
			\item Beide Ansätze streben nach Robustheit durch reduzierte Modellannahmen.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Zentrale Idee der robusten linearen Regression}
		
		\textbf{Zentrale Idee:}
		\begin{itemize}
			\itemsep8pt
			\item Anstatt eine globale Fehlerfunktion zu minimieren, werden Steigungen zwischen Beobachtungspaaren betrachtet.
			\item Die Schätzung basiert auf dem Median dieser Steigungen.
		\end{itemize}
		
		\vspace{0.5em}
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{Vergleich von Mittelwert und Median:}
		\begin{itemize}
			\itemsep8pt
			\item Mittelwert: Stark empfindlich gegenüber einzelnen Ausreißern.
			\item Median: Robust gegenüber extremen Beobachtungen und repräsentiert die zentrale Tendenz der Steigungen.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Theil-Schätzer – Einführung und Formel}
		
		Der \textbf{Theil-Sen-Schätzer} schätzt die Steigung $\beta$ eines linearen Modells $Y = \alpha + \beta X$ durch den Median der paarweisen Steigungen $H_{ij}$ zwischen allen Beobachtungspaaren $i < j$:
		\[
		H_{ij} = \frac{Y_j - Y_i}{X_j - X_i}, \qquad i < j.
		\]
		
		\vspace{0.5em}
		
		\ifpauseswitch
		\pause
		\fi
		
		Der Schätzer für den Anstieg ist dann:
		\[
		\hat{\beta} = \operatorname{Median}(H_{ij}).
		\]
		
		\textbf{Theil-Methode I:} Berechnet Steigungen nur zwischen Paaren $(i,\, i+n/2)$, wobei $n$ gerade ist und $i = 1,\dots, n/2$.
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Wichtige Eigenschaften des Theil-Sen-Schätzers}
		
		\begin{itemize}
			\itemsep10pt
			\item \textbf{Hoher Breakdown-Punkt:} Der Schätzer bleibt stabil, solange weniger als $50\,\%$ der Datenpunkte Ausreißer sind. Bei der klassischen linearen Regression reicht ein einziger Ausreißer, um die Schätzung stark zu verzerren.
			\item \textbf{Konsistenz:} Konsistent unter milden Voraussetzungen.
			\item \textbf{Anwendung:} Geeignet bei Datensätzen mit möglichen Ausreißern und in der explorativen Datenanalyse.
		\end{itemize}
	\end{frame}
	
	
	\begin{frame}
		\frametitle{Einschränkungen der robusten linearen Regression}
		
		\textbf{Einschränkungen:}
		\begin{itemize}
			\itemsep8pt
			\item Höherer Rechenaufwand bei großen Stichproben (Theil-Sen: $O(n^2)$ Steigungen).
			\item Beschränkt auf lineare Zusammenhänge zwischen $X$ und $Y$.
		\end{itemize}
		
		\vspace{0.5em}
		
		\ifpauseswitch
		\pause
		\fi
		
		\textbf{Abgrenzung zur nichtparametrischen Regression:}
		\begin{itemize}
			\itemsep8pt
			\item Die nichtparametrische Regression bietet höhere Flexibilität, da sie keine Linearität voraussetzt.
			\item Dafür ist die Interpretation der nichtparametrischen Regression weniger direkt.
		\end{itemize}
	\end{frame}
	
	
	\section{Zusammenfassung}
	
	\begin{frame}
		\frametitle{Zusammenfassung}
		\hypertarget{nachrobust}{}
		
		\begin{itemize}
			\itemsep10pt
			\item Nichtparametrische Methoden sind lokal und datengetrieben: Sie schätzen $f(x)$ oder $m(x)$ ohne Annahme einer parametrischen Familie.
			\item Der Kerndichteschätzer lässt sich als Faltung mit dem empirischen Maß auffassen.
			\item Die Bandbreite $h$ ist die zentrale Stellgröße und bestimmt die Konvergenzrate.
			\item Die nichtparametrische Regression folgt demselben Prinzip der lokalen Kerngewichtung.
			\item Robuste Verfahren wie der Theil-Sen-Schätzer bieten Schutz vor Ausreißern bei linearen Modellen.
		\end{itemize}
	\end{frame}
	
	\begin{frame}
		\frametitle{Fazit und Ausblick}
		
		\begin{itemize}
			\itemsep10pt
			\item \textbf{Flexibilität:} Nichtparametrische Methoden passen sich den Daten an, ohne eine spezifische funktionale Form vorauszusetzen.
			\item \textbf{Parameter $h$:} Die Wahl der Bandbreite ist der entscheidende Schritt und erfordert eine Balance zwischen Bias und Varianz.
			\item \textbf{Robustheit:} Verfahren wie der Theil-Sen-Schätzer bieten robuste Alternativen zur klassischen linearen Regression, besonders bei Vorliegen von Ausreißern.
		\end{itemize}
	\end{frame}
